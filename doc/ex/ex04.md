# ex04: Marge 2 Histograms
```
opt (fc:gray)       ;# fill area color gray
opt (rc:1)          ;# relative coordinate for text/fbox/symb position 
opt (cp:0.75,0.8)   ;# commnet position
opt (xt:0.1)        ;# x-axis tick
opt (nb:100 xr:0,2) ;# histogram nbin is 100, xrange is from 0 to 2
opt (ts:1.1)        ;# text scale

@ sigma=0.1
@ mean=0.7
@ tau=2
set v1 = random(10000,gaus,[sigma],[mean]) ;# random gaussian
set v2 = random(50000,exp,[tau])           ;# random exponential

div 1 3
xlab ""; ylab "Numebr of entries"

title "(A) Histogram of random Gaussian (sigma=[sigma], mean=[mean]) v1"
hplot v1 (ft:p45)
hfit v1 gaus (lc:green ts:0.9)

title "(B) Histogram of random Exponential (tau=[tau]) v2"
hplot v2 (ft:n45)
hfit v2 exp (lc:green ts:0.9)

title "(C) Histogram of concatenated distribution of v1 and v2, (D) Sum of entries of (A) and (B)"

cat v1,v2 > v3 ;# concatenate v1 and v2 to v3
hplot v3 (ft:pn45 xl:"v1+v2" yl:"v1+v2 entries")

set entry_sum = v1_hy + v2_hy
plot v1_hx entry_sum (lw:0 st:star sc:blue rp:1)
fbox 0.8 0.84 0.75 0.9 (lc:gray ft:pn45 lt:solid); text 0.85 0.85 "(C)"
symb 0.82 0.65 (st:star sc:blue);                  text 0.85 0.65 "(D)"
```
