# ex09: Data cutting and simplified logical expression
```
opt (lw:0)   ;# set line width 0 (not plot lines)
opt (ss:0.7) ;# set symbol size 
opt (xt:10)  ;# x-axis tick 10

@ ndata = 200
set x = range([ndata],0,99)
set y = random([ndata],uni,-1,1)

@ x0 = 30
@ x1 = 70
@ y0 = -0.5
@ y1 = 0.5

# cut conditions
@ cc1 = "([x0]<x && x<[x1]) && ([y0]<y && y<[y1])"
@ cc2 = "[x0]<x<[x1] & [y0]<y<[y1]"  ;# simple expression of cc1

div 1 2

do n 1 2
  title "cut x,y by the condition of [cc[n]]"
  plot x y (st:star)
  plot x y (cc:"[cc[n]]" st:arc sc:red rp:1)
  fbox [x0] [x1] [y0] [y1] (lt:dot lc:green)
end
```
